public class Vehicle {
    String brand;
    double price;
    void display() {
        System.out.println("Brand: " + brand);
        System.out.println("Price: " + price);
    }
    public static void main(String[] args) {
        Vehicle v1 = new Vehicle();
        v1.brand = "Tesla";
        v1.price = 50000.00;
        v1.display();
    }
}
